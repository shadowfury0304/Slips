Q2)

1. Print process Ids of process name bash:-   pgrep bash

2.Write shell script to accept the process id from user and forcefully kill the process with that id:-


read -p "Enter PID to forcefully kill: " PID

if [ -z "$PID" ] || ! [[ "$PID" =~ ^[0-9]+$ ]]; then
  echo "Invalid or empty PID provided."
  exit 1
fi

# Forcefully kill using signal 9 (SIGKILL)
kill -9 "$PID"

if [ $? -eq 0 ]; then
  echo "Process with PID $PID forcefully killed (SIGKILL)."
else
  echo "Failed to kill process $PID. Check permissions or if the PID exists."
fi
